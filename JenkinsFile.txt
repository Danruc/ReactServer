pipeline {
     agent any
     stages {
        stage("Build") {
            steps {
	    	sh "sudo npm install"
                sh "sudo npm install @mui/material @emotion/react @emotion/styled"
                sh "sudo npm install formik --save"
		sh "sudo npm install axios"
                sh "sudo npm run build"
            }
        }
        stage("Deploy") {
            steps {
                sh "sudo rm -rf /var/www/holaMundo"
                sh "sudo cp -r /var/lib/jenkins/workspace/holaMundo/build/ /var/www/holaMundo/"
            }
        }
    }
}
