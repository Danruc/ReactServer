pipeline {
     agent any
     stages {
        stage("Build") {
            steps {
                sh "npm install"
                sh "npm install @mui/material @emotion/react @emotion/styled"
                sh "npm install formik --save"
		sh "npm install axios"
                sh "npm run build"
            }
        }
        stage("Deploy") {
            steps {
                sh "rm -rf /var/www/NodeServer"
                sh "cp -r ${WORKSPACE}/build/ /var/www/NodeServer/"
            }
        }
    }
}
