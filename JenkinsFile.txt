pipeline {
     agent any
     stages {
        stage("Build") {
            steps {
	    	sh 'nodejs --version'
                sh 'npm install'
                sh 'gulp lint'
                
            }
        }
        stage("Deploy") {
            steps {
                sh "sudo rm -rf /var/www/NodeServer"
                sh "sudo cp -r ${WORKSPACE}/build/ /var/www/NodeServer/"
            }
        }
    }
}
